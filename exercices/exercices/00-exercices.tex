\textbf{\large E) ⚠️ Ambiguïtés de symboles — À bien distinguer}

\noindent
\fcolorbox{FedRed}{FedRed!5}{%
  \begin{minipage}{0.95\textwidth}\small
    \textbf{$P$ a deux sens distincts :}
    \begin{itemize}[leftmargin=1.5em, itemsep=0.2em]
      \item $P_{\text{hyd}}$ ou $P$ en équations thermiques = \textbf{Puissance} en W (watts)
      \item $P$ dans formules de dynamique (ex: $c = \sqrt{\gamma P / \rho}$) = \textbf{Pression} en Pa
    \end{itemize}
    \textbf{→ Toujours vérifier le contexte et les unités !}
  \end{minipage}
}

  \item \textbf{D — Manipulation algébrique :} Isoler la variable cible, toujours vérifier dimensions finales
  \item \textbf{E — Non-linéarité :} Racines/puissances peuvent donner 2 solutions \text{; analyser physiquement}
  \item \textbf{F — Masse vs Volume :} Relation $\rho = \dot{m}/\dot{V}$ \textbf{fondamentale} en FED
  \item \textbf{G — Analyse dimensionnelle :} Système 3 équations / 4 inconnues $\to$ 1 degré de liberté
  \item \textbf{H — Ordre de grandeur :} \textbf{TOUJOURS} valider : résultat plausible physiquement ?
  \item \textbf{I — Interpolation :} Linéaire par défaut ; log-log si donnees couvrent plusieurs ordres
  \item \textbf{J — Synthèse :} Combine 3--4 techniques ; exercices proches du terrain industrie
\end{itemize}
