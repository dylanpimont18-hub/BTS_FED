\section{Objectifs pédagogiques}
% ============================================================

À l'issue de ce chapitre, l'étudiant sera capable de :
\begin{itemize}
  \item Conduire une démarche complète de dimensionnement CVC en partant du cahier des charges ;
  \item Calculer les déperditions thermiques d'un bâtiment selon la méthode réglementaire (RE2020 / NF EN 12831) ;
  \item Sélectionner et dimensionner les équipements de production, distribution et émission de chaleur ;
  \item Dimensionner un réseau aéraulique et une centrale de traitement d'air (CTA) ;
  \item Appliquer les critères d'efficacité énergétique et vérifier la conformité réglementaire ;
  \item Intégrer la régulation et la GTB dans une étude globale ;
  \item Rédiger une note de calcul justifiée à présenter en situation professionnelle.

\section{Méthode générale de dimensionnement}
% ============================================================

\subsection{Démarche systématique}

Le technicien supérieur GCF est régulièrement amené à produire des études techniques complètes : appels d'offres, avant-projets sommaires (APS), avant-projets définitifs (APD), dossiers de consultation des entreprises (DCE). Ces études intègrent plusieurs disciplines enseignées séparément dans les chapitres précédents.

\begin{methode}[title={Étapes du dimensionnement CVC}]
  \begin{enumerate}
    \item \textbf{Collecte des données} : plans architectural, orientation, localisation climatique, destination des locaux, réglementation applicable.
    \item \textbf{Bilan thermique d'hiver} : calcul des déperditions par transmission et renouvellement d'air (NF EN 12831).
    \item \textbf{Bilan thermique d'été} : calcul des charges de refroidissement (apports solaires, internes, renouvellement).
    \item \textbf{Sélection des générateurs} : chaudière, PAC, groupe froid — selon besoins de pointe et modulabilité.
    \item \textbf{Distribution} : réseau hydraulique (dimensionnement des tuyauteries, pompes, équilibrage) et/ou aéraulique (gaines, ventilateurs, diffuseurs).
    \item \textbf{Émission} : radiateurs, planchers chauffants, ventilo-convecteurs, soufflage.
    \item \textbf{Régulation et GTB} : stratégies de commande, sondes, actionneurs.
    \item \textbf{Vérification RE2020} : calcul Bbio, Cep, DH.
    \item \textbf{Rédaction de la note de calcul} et sélection du matériel sur catalogue.
  \end{enumerate}
\end{methode}

\begin{definition}[title={Note de calcul}]
  La note de calcul est le document technique central d'une étude CVC. Elle regroupe :
  les données d'entrée (géométrie, occupation, conditions climatiques), les hypothèses
  retenues, les formules appliquées avec leurs références normatives, les résultats
  numériques et la sélection du matériel. Elle doit être \textbf{traçable, vérifiable et justifiée}.
\end{definition}

\subsection{Notion de pincement thermique}

\begin{definition}[title={Pincement thermique (temperature pinch)}]
  Le \textbf{pincement thermique} (ou \emph{pinch}) désigne l'écart minimal de température
  entre deux fluides échangeant de la chaleur dans un échangeur ou dans un cycle thermodynamique.
  C'est le point où les températures des deux fluides sont les plus proches l'une de l'autre.

  \medskip
  \textbf{Formule générale :}
  \begin{equation}
    \Delta T_{pinch} = T_{fluide\_chaud,min} - T_{fluide\_froid,max} \quad [\si{\kelvin}]
  \end{equation}

  \textbf{Valeurs courantes en CVC :}
  \begin{itemize}
    \item Batterie froide eau glacée / air : $\Delta T_{pinch} = 3$ à $\SI{6}{\celsius}$
    \item Condenseur groupe froid / eau de tour : $\Delta T_{pinch} = 4$ à $\SI{8}{\celsius}$
    \item Échangeur récupération chaleur VMC : $\Delta T_{pinch} = 2$ à $\SI{4}{\celsius}$
    \item Évaporateur PAC géothermique : $\Delta T_{pinch} = 3$ à $\SI{5}{\celsius}$
  \end{itemize}

  \textbf{Importance :} Un pincement faible améliore le transfert thermique mais augmente
  la surface d'échange nécessaire (coût). Un pincement trop élevé dégrade l'efficacité du
  cycle (COP réduit). Le choix du pincement résulte d'un compromis technico-économique.
\end{definition}

\begin{attention}
  En pratique, le pincement thermique fixe la température minimale d'eau glacée requise.
  Exemple : pour refroidir de l'air à $T_{soufflage} = \SI{15}{\celsius}$, avec un
  pincement de $\SI{4}{\celsius}$, la température d'entrée eau glacée doit être
  $\leq 15 - 4 = \SI{11}{\celsius}$. On retient couramment le circuit \SI[parse-numbers=false]{7/12}{\celsius}.
\end{attention}

\subsection{Rappel des formules fondamentales}

\subsubsection{Déperditions par transmission}

\begin{equation}
  \Phi_{T} = H_T \cdot (T_{int} - T_{ext,base})
\end{equation}

avec $H_T = \sum_{i} U_i \cdot A_i$ le coefficient de déperdition global par transmission (\si{\watt\per\kelvin}).

\subsubsection{Déperditions par ventilation et infiltrations}

\begin{equation}
  \Phi_{V} = H_V \cdot (T_{int} - T_{ext,base})
\end{equation}

avec $H_V = 0{,}34 \times q_{v,m^3/h} \times (1 - \eta_{recup})$ (\si{\watt\per\kelvin}), où $q_v$ est le débit volumique d'air neuf.

\subsubsection{Puissance de chauffage de pointe}

\begin{equation}
  \Phi_{ch} = (H_T + H_V) \cdot (T_{int} - T_{ext,base}) \quad [\si{\watt}]
\end{equation}

\subsubsection{Coefficient de performance de la PAC}

\begin{equation}
  \mathrm{COP} = \frac{P_{calorifique}}{P_{electrique}}, \quad
  \mathrm{SCOP} = \frac{W_{calorifique,annuel}}{W_{electrique,annuel}}
\end{equation}

\subsubsection{Consommation annuelle PAC (méthode DJU)}

\begin{equation}
  W_{el} = \frac{\Phi_{ch} \times DJU \times 24}{SCOP \times \Delta T_{base}} \quad [\si{\kilo\watt\hour\per\year}]
\end{equation}

\subsubsection{Indicateur d'énergie primaire RE2020}

\begin{equation}
  Cep = \frac{W_{el} \times f_{prim}}{A_{ref}} \quad [\si{\kilo\watt\hour\per\metre\squared\per\year}_{\mathrm{ep}}], \quad f_{prim,elec} = 2{,}3
\end{equation}

\subsubsection{Perte de charge en réseau hydraulique}

\begin{equation}
  \Delta P = \lambda \cdot \frac{L}{D} \cdot \frac{\rho v^2}{2} + \sum \xi_i \cdot \frac{\rho v^2}{2} \quad [\si{\pascal}]
\end{equation}

\subsubsection{Puissance frigorifique nécessaire}

\begin{equation}
  Q_f = Q_{apports\_solaires} + Q_{apports\_internes} + Q_{ventilation}
\end{equation}

\subsubsection{Critères de sélection des équipements}

\begin{center}
\begin{tabular}{lll}
  \toprule
  Équipement & Critère principal & Critère secondaire \\
  \midrule
  PAC air/eau & COP $\geq 3{,}2$ à conditions nominales & SCOP annuel, niveau acoustique \\
  Groupe froid & EER $\geq 2{,}8$ (classe A), GWP fluide $<$ 750 & Régulation, free-cooling intégré \\
  CTA double flux & Rendement échangeur $\geq 75\,\%$ (SFP$\leq$1\,W/(m\textsuperscript{3}/h)) & Compacité, niveaux filtration \\
  Pompe circulatrice & Point de fonctionnement $\subset$ courbe de pompe & Efficacité hydraulique ($\eta \geq 65\,\%$) \\
  Ventilateur & $\eta_{tot} \geq 70\,\%$, classe IE3 & Niveau sonore $\leq 45$ dB(A) \\
  \bottomrule
\end{tabular}
\end{center}

\section{Étude de cas 1 — Maison individuelle (PAC + plancher chauffant)}
% ============================================================

\subsection{Présentation du projet}

Maison individuelle de plain-pied construite à \textbf{Lyon} (zone H2b, $T_{ext,base} = \SI{-8}{\celsius}$,
$DJU = 2\,400$). Surface habitable : \SI{120}{\metre\squared}, hauteur sous plafond \SI{2{,}5}{\metre}.
Système CVC : PAC air/eau alimentant un plancher chauffant basse température, VMC double flux,
conformité RE2020 exigée.

\textbf{Objectifs de l'étude :} calculer la puissance de pointe, sélectionner la PAC,
dimensionner le plancher chauffant, vérifier l'indicateur Cep RE2020.

\subsection{Théorie — Calcul des besoins thermiques}

Le bilan thermique d'hiver s'appuie sur la norme NF EN 12831 :

\begin{equation}
  \Phi_{ch} = (H_T + H_V) \cdot (T_{int} - T_{ext,base})
\end{equation}

où $H_T = \sum U_i A_i$ cumule les déperditions par toutes les parois opaques et vitrées, et
$H_V$ intègre l'effet du récupérateur de chaleur de la VMC double flux.

La \textbf{puissance surfacique} du plancher chauffant doit respecter les contraintes RE2020 :
\begin{equation}
  q_{plancher} = \frac{\Phi_{ch}}{A_{totale}} \leq \SI{60}{\watt\per\metre\squared}
\end{equation}

La sélection de la PAC impose de vérifier le \textbf{pincement thermique} dans l'échangeur
eau/réfrigérant : pour un plancher chauffant à \SI{35}{\celsius}, la PAC doit délivrer
l'eau à cette température même à $T_{ext,base}$, ce qui dégrade le COP par rapport aux
conditions nominales A7/W35. Le fabricant doit fournir les courbes de performance à $-8/35\,\si{\celsius}$.

\begin{exemple}[title={Maison 120 m² à Lyon — Bilan complet}]
\textbf{Données enveloppe :}

\begin{center}
\begin{tabular}{llccc}
  \toprule
  Paroi & Description & $A$ (\si{\metre\squared}) & $U$ (\si{\watt\per\metre\squared\per\kelvin}) & $UA$ (\si{\watt\per\kelvin}) \\
  \midrule
  Murs extérieurs    & ITE laine de roche 14 cm & 115 & 0,22 & 25,3 \\
  Toiture            & Combles perdus 35 cm     & 120 & 0,13 & 15,6 \\
  Plancher bas       & Dalle isolée 8 cm        & 120 & 0,32 & 38,4 \\
  Vitrages           & Double vitrage $U_w=1{,}2$ &  25 & 1,20 & 30,0 \\
  Portes extérieures & Porte isolée             &   4 & 1,40 &  5,6 \\
  \midrule
  \textbf{Total $H_T$} & & & & \textbf{114,9} \\
  \bottomrule
\end{tabular}
\end{center}

\textbf{Déperditions par ventilation (VMC double flux, $\eta_{recup} = 0{,}85$) :}
\begin{align*}
  V_{bat} &= 120 \times 2{,}5 = \SI{300}{\metre\cubed} \\
  q_v &= 0{,}40\,\mathrm{vol/h} \times 300 = \SI{120}{\metre\cubed\per\hour} \\
  H_V &= 0{,}34 \times 120 \times (1 - 0{,}85) = \SI{6{,}1}{\watt\per\kelvin}
\end{align*}

\textbf{Puissance de pointe ($\Delta T = 19 - (-8) = \SI{27}{\kelvin}$) :}
\begin{align*}
  H_{total} &= 114{,}9 + 6{,}1 = \SI{121{,}0}{\watt\per\kelvin} \\
  \Phi_{ch}  &= 121{,}0 \times 27 = \SI{3\,267}{\watt} \approx \SI{3{,}3}{\kilo\watt}
\end{align*}

\textbf{Sélection de la PAC (marge 15\,\%) :}
\begin{equation*}
  P_{PAC,installée} = 3{,}3 \times 1{,}15 = \SI{3{,}8}{\kilo\watt} \rightarrow \textbf{PAC 4 kW sélectionnée}
\end{equation*}
COP retenu à $-8/35$ : $COP = 3{,}0$ (courbe fabricant). \\
Puissance électrique absorbée en pointe : $P_{el} = 3{,}3 / 3{,}0 = \SI{1{,}1}{\kilo\watt}$.

\textbf{Dimensionnement plancher chauffant :}
\begin{align*}
  q_{plancher} &= \frac{3\,267}{120} = \SI{27{,}2}{\watt\per\metre\squared}
\end{align*}
Valeur très inférieure à la limite RE2020 de \SI{60}{\watt\per\metre\squared}.
Température de départ eau retenue : \SI{35}{\celsius} (retour \SI{30}{\celsius}).

\textbf{Pincement thermique à la batterie PAC :}
$\Delta T_{pinch} = 35 - 30 = \SI{5}{\celsius}$ côté eau ;
côté réfrigérant, condensation à $\approx \SI{40}{\celsius}$ — cohérent avec les données fabricant.

\textbf{Consommation annuelle et vérification RE2020 (SCOP = 4,0) :}
\begin{align*}
  W_{el} &= \frac{3{,}3 \times 2\,400 \times 24}{4{,}0 \times 27} = \SI{1\,760}{\kilo\watt\hour\per\year} \\
  Cep    &= \frac{1\,760 \times 2{,}3}{120} = \SI{33{,}7}{\kilo\watt\hour\per\metre\squared\per\year}_{\mathrm{ep}}
\end{align*}

\textbf{Conclusion :} $33{,}7 < 70\,\si{\kilo\watt\hour\per\metre\squared\per\year}_{\mathrm{ep}}$ — \textbf{conforme RE2020}.
Excellent niveau de performance, proche des exigences BEPOS.
\end{exemple}

\section{Étude de cas 2 — Bâtiment tertiaire (CTA + refroidissement)}
% ============================================================

\subsection{Présentation du projet}

Plateau de bureaux paysagers de \textbf{500\,m²} (hauteur \SI{3}{\metre}, 50 personnes).
Localisation : \textbf{Bordeaux} (zone H2c, $T_{ext,base} = \SI{-3}{\celsius}$).
Conditions d'été : $T_{ext} = \SI{34}{\celsius}$, HR = 50\,\%.
Système CVC : CTA double flux avec batterie froide, groupe d'eau glacée,
free-cooling aéraulique, GTB.

\textbf{Objectifs :} bilan aéraulique et frigorifique, sélection CTA, vérification RE2020 tertiaire.

\subsection{Théorie — Bilan aéraulique et free-cooling}

La charge de refroidissement d'un bâtiment tertiaire comprend quatre postes :
\begin{equation}
  Q_{f,total} = Q_{int} + Q_{solaire} + Q_{vent} + Q_{parois}
\end{equation}

où $Q_{parois}$ est souvent négligé en bâtiments bien isolés.

Le \textbf{free-cooling aéraulique} consiste à utiliser l'air extérieur frais ($T_{ext} < T_{int} - 2\,\si{\celsius}$)
comme source de refroidissement gratuit, réduisant la sollicitation du groupe froid.
La plage de free-cooling à Bordeaux est estimée à environ 2\,500\,h/an.

Le \textbf{pincement thermique dans la batterie froide} conditionne la température d'eau glacée :
pour une température de soufflage de \SI{15}{\celsius}, avec un pincement de \SI{4}{\celsius},
la température d'eau glacée en entrée doit être $\leq \SI{11}{\celsius}$ ;
on choisit le circuit standard \SI[parse-numbers=false]{7/12}{\celsius}.

\begin{exemple}[title={Bureaux 500 m² à Bordeaux — Dimensionnement CTA}]
\textbf{Données :}
\begin{itemize}
  \item Débit d'air hygiénique : $25\,\si{\litre\per\second}/pers \times 50\,pers = \SI{1\,250}{\litre\per\second} = \SI{4\,500}{\metre\cubed\per\hour}$
  \item Apports internes : personnes \SI{70}{\watt}/pers, éclairage \SI{8}{\watt\per\metre\squared}, bureautique \SI{20}{\watt\per\metre\squared}
  \item Apports solaires : vitrages Sud \SI{80}{\metre\squared}, $g = 0{,}32$, $G_{solaire} = \SI{550}{\watt\per\metre\squared}$
  \item Température intérieure : \SI{24}{\celsius} ; température de soufflage : \SI{15}{\celsius}
  \item Perte de charge réseau aéraulique : \SI{600}{\pascal} ; rendement ventilateur : $\eta_{vent} = 0{,}72$
\end{itemize}

\textbf{1 — Apports internes :}
\begin{align*}
  Q_{personnes}  &= 50 \times 70 = \SI{3\,500}{\watt} \\
  Q_{eclairage}  &= 500 \times 8 = \SI{4\,000}{\watt} \\
  Q_{bureautique}&= 500 \times 20 = \SI{10\,000}{\watt} \\
  Q_{int,total}  &= \SI{17\,500}{\watt} = \SI{17{,}5}{\kilo\watt}
\end{align*}

\textbf{2 — Apports solaires :}
\begin{align*}
  Q_{solaire} &= g \times A_{vitrage} \times G_{solaire} = 0{,}32 \times 80 \times 550 = \SI{14\,080}{\watt} \approx \SI{14{,}1}{\kilo\watt}
\end{align*}

\textbf{3 — Charge de renouvellement d'air ($\Delta T = 34 - 24 = \SI{10}{\kelvin}$) :}
\begin{align*}
  Q_{vent} &= \frac{1{,}2 \times 1005 \times 4\,500}{3\,600} \times 10 = 1\,507 \times 10 = \SI{15\,070}{\watt} \approx \SI{15{,}1}{\kilo\watt}
\end{align*}

\textbf{4 — Charge totale et sélection groupe froid (marge 10\,\%) :}
\begin{align*}
  Q_{f,total}     &= 17{,}5 + 14{,}1 + 15{,}1 = \SI{46{,}7}{\kilo\watt} \\
  Q_{GF,installé} &= 46{,}7 \times 1{,}10 = \SI{51{,}4}{\kilo\watt} \rightarrow \textbf{groupe froid 55 kW}
\end{align*}

\textbf{5 — Pincement thermique batterie froide :}
$T_{soufflage} = \SI{15}{\celsius}$, $\Delta T_{pinch} = \SI{4}{\celsius}$ \\
$\Rightarrow T_{EG,entrée} \leq 15 - 4 = \SI{11}{\celsius}$ — circuit \SI[parse-numbers=false]{7/12}{\celsius} retenu.

\textbf{6 — Puissance batterie froide CTA :}
\begin{align*}
  Q_{batterie} &= \frac{1{,}2 \times 1005 \times 4\,500}{3\,600} \times (24 - 15) = 1\,507 \times 9 = \SI{13\,560}{\watt} \approx \SI{13{,}6}{\kilo\watt}
\end{align*}

\textbf{7 — Débit eau glacée batterie :}
\begin{align*}
  \dot{V}_{EG} &= \frac{Q_{batterie}}{\rho \cdot c_p \cdot \Delta T_{EG}} = \frac{13\,560}{1\,000 \times 4\,186 \times 5 / 3\,600} = \frac{13\,560}{5\,814} = \SI{2{,}33}{\metre\cubed\per\hour}
\end{align*}

\textbf{8 — Puissance du ventilateur de soufflage :}
\begin{align*}
  P_{vent} &= \frac{q_v \times \Delta P_{tot}}{\eta_{vent}} = \frac{(4\,500/3\,600) \times 600}{0{,}72} = \frac{1{,}25 \times 600}{0{,}72} = \SI{1\,042}{\watt} \approx \SI{1{,}0}{\kilo\watt}
\end{align*}

\textbf{9 — Puissance électrique groupe froid ($EER = 3{,}2$) :}
\begin{align*}
  P_{GF,el} &= \frac{Q_{GF,installé}}{EER} = \frac{55}{3{,}2} = \SI{17{,}2}{\kilo\watt}
\end{align*}

\textbf{10 — Bilan énergétique et RE2020 tertiaire :}

Consommation annuelle groupe froid (2\,000\,h équivalent plein régime, dont 500\,h free-cooling) :
\begin{align*}
  W_{GF,annuel} &= 17{,}2 \times (2\,000 - 500) = \SI{25\,800}{\kilo\watt\hour\per\year} \\
  Cep_{refroid} &= \frac{25\,800 \times 2{,}3}{500} = \SI{118{,}7}{\kilo\watt\hour\per\metre\squared\per\year}_{\mathrm{ep}}
\end{align*}

Seuil RE2020 tertiaire (décret tertiaire, objectif 2030) : réduction 40\,\% /2010.
Le free-cooling réduit la consommation de 25\,\% environ — action prioritaire à valoriser.
\end{exemple}

\section{Étude de cas 3 — Industrie pharmaceutique (salle propre)}

\subsection{Présentation du projet}

Atelier de production pharmaceutique à \textbf{Strasbourg} (zone H1c, $T_{ext,base} = \SI{-11}{\celsius}$).
Surface : \SI{600}{\metre\squared}, hauteur \SI{3}{\metre}.
Exigences : classification \textbf{ISO\,7}, température $22 \pm \SI{1}{\celsius}$,
HR $50 \pm 5\,\%$, surpression $+15\,\si{\pascal}$ par rapport aux zones adjacentes,
renouvellement d'air conforme BPF (20 volumes/heure minimum).
Énergie disponible sur site : vapeur process à \SI{6}{\bar}, eau glacée \SI[parse-numbers=false]{6/12}{\celsius}.

\textbf{Objectifs :} classification ISO 7, dimensionnement CTA HEPA, bilan énergétique annuel,
vérification des exigences de surpression et contamination.

\subsection{Théorie — Classification ISO et gestion de la contamination}

\begin{definition}[title={Classification ISO des salles propres (ISO 14644-1)}]
  La classification ISO définit le nombre maximal de particules admissibles par m\textsuperscript{3} d'air.

  \begin{center}
  \begin{tabular}{lcc}
    \toprule
    Classe ISO & Particules $\geq 0{,}5\,\si{\micro\metre}$ / m\textsuperscript{3} & Taux de brassage indicatif \\
    \midrule
    ISO 5  & $3\,520$       & 240 -- 480 vol/h \\
    ISO 6  & $35\,200$      & 90 -- 180 vol/h \\
    ISO 7  & $352\,000$     & 30 -- 60 vol/h \\
    ISO 8  & $3\,520\,000$  & 10 -- 25 vol/h \\
    \bottomrule
  \end{tabular}
  \end{center}

  La \textbf{surpression} (de $+10$ à $+25\,\si{\pascal}$) empêche les contaminations
  extérieures de pénétrer dans la salle propre. La filtration HEPA H14 retient
  $\geq 99{,}995\,\%$ des particules à $0{,}3\,\si{\micro\metre}$.
\end{definition}

Le \textbf{pincement thermique dans la batterie froide} est particulièrement critique en
salle propre : la batterie doit refroidir l'air jusqu'à sa température de rosée pour
assurer la déshumidification, puis le réchauffer jusqu'à la température de soufflage.
Ce processus en deux étapes impose un pincement minimum de $\SI{3}{\celsius}$ entre la
température de rosée cible et la température d'eau glacée.

\textbf{Température de rosée cible} (ISO\,7, $22\,\si{\celsius}$, $HR = 50\,\%$) :
$T_{rosée} \approx \SI{10{,}8}{\celsius}$.
Avec pincement de $\SI{3}{\celsius}$, température eau glacée entrée $\leq \SI{7{,}8}{\celsius}$ :
le circuit \SI[parse-numbers=false]{6/12}{\celsius} disponible sur site convient.

\textbf{Débit d'air neuf (20 vol/h minimum BPF) :}
\begin{align*}
  q_{v,AN} &= 20 \times 600 \times 3 = \SI{36\,000}{\metre\cubed\per\hour}
\end{align*}
Vérification ISO\,7 (taux de brassage 30 à 60 vol/h) : $36\,000 / (600 \times 3) = 20\,\mathrm{vol/h}$ —
on retient \textbf{40 vol/h} pour garantir ISO\,7, soit
$q_{v,total} = 40 \times 1\,800 = \SI{72\,000}{\metre\cubed\per\hour}$.

\textbf{Charge frigorifique totale (été, $T_{ext} = \SI{33}{\celsius}$) :}
\begin{align*}
  Q_{vent} &= \frac{1{,}2 \times 1005 \times 36\,000}{3\,600} \times (33 - 22) = 12\,060 \times 11 = \SI{132\,660}{\watt} = \SI{132{,}7}{\kilo\watt} \\
  Q_{f,total} &= 50{,}9 + 132{,}7 = \SI{183{,}6}{\kilo\watt}
\end{align*}

Avec marge de sécurité 15\,\% : $Q_{GF} = 183{,}6 \times 1{,}15 \approx \SI{211}{\kilo\watt}$ — sélection \textbf{groupe froid 220\,kW}.

\textbf{Puissance électrique groupe froid ($EER = 3{,}0$ à $6/12\,\si{\celsius}$) :}
\begin{align*}
  P_{GF,el} &= \frac{220}{3{,}0} = \SI{73{,}3}{\kilo\watt}
\end{align*}

\textbf{Récupération chaleur sur air extrait ($\eta = 0{,}75$, $T_{ext,hiver} = \SI{-11}{\celsius}$) :}
\begin{align*}
  Q_{recup} &= 0{,}75 \times \frac{1{,}2 \times 1005 \times 36\,000}{3\,600} \times (22 - (-11)) \\
             &= 0{,}75 \times 12\,060 \times 33 = \SI{298\,485}{\watt} \approx \SI{299}{\kilo\watt}
\end{align*}

\textbf{Bilan énergétique annuel (3\,200\,h/an fonctionnement) :}
\begin{align*}
  W_{GF,annuel}    &= 73{,}3 \times 3\,200 = \SI{234\,560}{\kilo\watt\hour\per\year} \\
  W_{recup,annuel} &= 299 \times 3\,200 \times 10^{-3} = \SI{956\,800}{\kilo\watt\hour\per\year} \text{ (économie chauffage)}
\end{align*}

\textbf{Vérification surpression et contamination :}
\begin{itemize}
  \item Débit de fuite estimé (joints, sas) : $q_{fuite} \approx 5\,\%$ du débit total $= \SI{3\,600}{\metre\cubed\per\hour}$
  \item Débit de compensation nécessaire : inclus dans le débit AN de \SI{36\,000}{\metre\cubed\per\hour}
  \item Surpression maintenue à $+15\,\si{\pascal}$ par régulation du volet d'extraction
  \item Filtration H14 : efficacité $\geq 99{,}995\,\%$ à $0{,}3\,\si{\micro\metre}$ — conforme ISO\,7
\end{itemize}

\textbf{Conclusion :} L'installation est dimensionnée pour maintenir les exigences ISO\,7 et BPF.
La récupération thermique permet d'économiser l'équivalent de $\approx \SI{957}{\mega\watt\hour\per\year}$
en chauffage et humidification, représentant un retour sur investissement estimé à 4 ans.
\end{exemple}

\section{Méthode de vérification et bilan global}

\subsection{Check-list de vérification finale}

À l'issue de tout dimensionnement CVC, une vérification systématique est indispensable
avant de finaliser la note de calcul. La check-list ci-dessous couvre les domaines RE2020,
acoustique et hydraulique.

\begin{methode}[title={Check-list de vérification — Note de calcul CVC}]
\textbf{Réglementation thermique RE2020 :}
\begin{itemize}
  \item[$\square$] $Bbio \leq Bbio_{max}$ vérifié (calcul RE2020 via logiciel agréé) ?
  \item[$\square$] $Cep \leq Cep_{max}$ vérifié (coefficient de conversion électricité $f_{prim} = 2{,}3$) ?
  \item[$\square$] $Cep,nr \leq Cep,nr_{max}$ vérifié (part non renouvelable) ?
  \item[$\square$] $DH \leq 1\,250$ degrés-heures d'inconfort d'été (résidentiel) ?
  \item[$\square$] Icénergie $\leq$ seuil progressif (2022, 2025, 2028, 2031) ?
  \item[$\square$] Iconstruction vérifié pour les matériaux de structure et isolation ?
\end{itemize}

\textbf{Acoustique :}
\begin{itemize}
  \item[$\square$] Niveau de pression acoustique des équipements (PAC, CTA, groupes froids) respecté ($\leq$ 60 dB(A) à 5\,m en zone résidentielle) ?
  \item[$\square$] Niveau de bruit dans les locaux respecté (NR25 bureaux, NR35 locaux industriels) ?
  \item[$\square$] Silencieux en gaine prévus aux départs de la CTA ?
  \item[$\square$] Liaisons souples sur toutes les connexions équipements/gaines et tuyauteries ?
\end{itemize}

\textbf{Hydraulique :}
\begin{itemize}
  \item[$\square$] Point de fonctionnement des pompes situé dans la plage optimale ($\eta \geq 65\,\%$) ?
  \item[$\square$] Équilibrage du réseau hydraulique (robinets d'équilibrage, régulation $\Delta P$) ?
  \item[$\square$] Vase d'expansion correctement dimensionné (volume, pression de tarage) ?
  \item[$\square$] Vitesses d'eau dans les tuyauteries respectées (0,5 à 1,5\,m/s logements, 1 à 2\,m/s tertiaire) ?
  \item[$\square$] Dispositifs de sécurité présents (soupape de sécurité, disconnecteur, purges) ?
\end{itemize}

\textbf{Aéraulique :}
\begin{itemize}
  \item[$\square$] Vitesses d'air dans les gaines respectées ($\leq$ 5\,m/s zones occupées) ?
  \item[$\square$] Facteur de transport SFP $\leq 1{,}5\,\si{\watt\per\metre\cubed\per\hour}$ (RT2012/RE2020) ?
  \item[$\square$] Débits réglementaires d'air neuf vérifiés (arrêté du 24 mars 1982) ?
  \item[$\square$] Filtration adaptée à l'usage (G4 minimum, M5 en tertiaire, H14 en salle propre) ?
\end{itemize}

\textbf{Fluides frigorigènes et sécurité :}
\begin{itemize}
  \item[$\square$] Fluide frigorigène à GWP $\leq 750$ (F-gas 2024 pour équipements neufs) ?
  \item[$\square$] Charge de fluide déclarée et traçabilité assurée ?
  \item[$\square$] Détecteur de fuite et soupape de sécurité présents ($> 5\,\mathrm{t\,CO_2eq}$) ?
\end{itemize}
\end{methode}
